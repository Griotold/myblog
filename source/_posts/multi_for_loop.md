### 이중 for문으로 전체 구구단표 만들기

for a in range(2, 10):
  for b in range(1, 10):
    c = a * b
    print('%d x %d = %d' % (a, b, c))

  print('-' * 50)

### 실습1
- 위의 별을 좌우반전으로 찍어보자.

  for j in range(1, i+1):

  print()
